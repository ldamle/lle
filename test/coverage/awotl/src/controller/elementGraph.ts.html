import * as Interface from '../interface';
import * as Types from '../types';
import Factories from '../factories';
 
class ElementGraph implements Interface.ElementGraph {
    tree: Types.ElementGraphNode[];
 
    constructor(pointElement: Interface.Element) {
        this.tree = [];
        this.genGraph(pointElement);
    }
 
    genGraph(pointElement: Interface.Element): void {
        const visited = new Set<Interface.Element>();
        const rootNode = this.buildNode(pointElement, visited);
        if (rootNode) {
            this.tree.push(rootNode);
        }
    }
 
    private buildNode(element: Interface.Element, visited: Set<Interface.Element>): Types.ElementGraphNode | null {
        // Проверяем если элемент уже был посещен
        if (visited.has(element)) {
            return null;
        }
 
        visited.add(element);
 
        const node: Types.ElementGraphNode = {
            element: element,
            connection: element.out_connections,
            out: []
        };
 
        // Проходим по всем выходным соединениям
        for (let i = 0; i < element.out_connections.length; i++) {
            const conn = element.out_connections[i];
 
            if (conn.out) {
                const nextElement = conn.out.element;
 
                // Проверяем если следующий элемент является генератором
                if ('in_connections' in nextElement) {
                    const childNode = this.buildNode(nextElement, visited);
                    if (childNode) {
                        node.out.push(childNode);
                    }
                } else {
                    // Это генератор, добавляем напрямую в дерево
                    this.tree.push({
                        element: nextElement,
                        connection: [],
                        out: []
                    });
                }
            }
        }
 
        return node;
    }
 
    findGenerators(pointElement: Interface.Element): Types.ElementGraphNode[] {
        if ("in_connections" in (pointElement as object)) {
            for (let i = 0; i < (pointElement as any).in_connections.length; i++) {
                this.findGenerators((pointElement as any).in_connections[i].out.element)
            }
            for (let i = 0; i < pointElement.out_connections.length; i++) {
                if (pointElement.out_connections[i].in) {
                    for (let j = 0; j < pointElement.out_connections.length; j++) {
                        this.findGenerators(((pointElement.out_connections[i].in as Types.SourcesArray)[j]).element)
                    }
                }
            }
        } else {
            this.tree.push({ element: pointElement, connection: [], out: [] });
        }
        return this.tree
    };
